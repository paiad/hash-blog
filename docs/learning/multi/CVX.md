            ADL = (p(k) * abs(h_total)^2) / (log(2) * (sigma2 + P_d) * s_d^2);
            BDL = -2 * (x_b - x0(k)) * ADL;
            CDL = -2 * (y_b - y0(k)) * ADL;

            BEL2 = -2 * (x_e2 - x0(k)) * AEL2;
            BEL3 = -2 * (x_e3 - x0(k)) * AEL3;
            CEL = -2 * (y_e - y0(k)) * AEL;
            CEL2 = -2 * (y_e2 - y0(k)) * AEL2;
            CEL3 = -2 * (y_e3 - y0(k)) * AEL3;

        end

            CDL = -2 * (y_b - y0(k)) * ADL;
